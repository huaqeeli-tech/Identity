CREATE DATABASE `trainingdb` /*!40100 DEFAULT CHARACTER SET utf8 COLLATE utf8_bin */;
CREATE TABLE `coursesdata` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `MILITARYID` varchar(45) DEFAULT NULL,
  `COURSID` varchar(45) DEFAULT NULL,
  `COURSNUMBER` varchar(45) DEFAULT NULL,
  `COURSPLASE` varchar(45) DEFAULT NULL,
  `COURSDURATION` varchar(200) DEFAULT NULL,
  `STARTDATE` varchar(45) DEFAULT NULL,
  `ENDDATE` varchar(45) DEFAULT NULL,
  `COURSIMAGE` longblob,
  `COURSESTIMATE` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`ID`)
) ENGINE=InnoDB AUTO_INCREMENT=14 DEFAULT CHARSET=utf8;
CREATE TABLE `coursnames` (
  `COURSID` int(11) NOT NULL AUTO_INCREMENT,
  `CORSNAME` varchar(200) DEFAULT NULL,
  PRIMARY KEY (`COURSID`)
) ENGINE=InnoDB AUTO_INCREMENT=1012 DEFAULT CHARSET=utf8;
CREATE TABLE `personaldata` (
  `MILITARYID` varchar(45) NOT NULL,
  `PERSONALID` varchar(45) DEFAULT NULL,
  `NAME` varchar(300) DEFAULT NULL,
  `RANK` varchar(45) DEFAULT NULL,
  `UNIT` varchar(200) DEFAULT NULL,
  PRIMARY KEY (`MILITARYID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
